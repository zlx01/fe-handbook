js
const uniq = require('uniq')

let arr1 = [12,12,25,14,18,68,25,25]
console.log(uniq(arr1)) // [ 12, 14, 18, 25, 68 ]

let arr2 = [1,11,2,3,3,12,11,12,13,13];
console.log(uniq(arr2)) // [ 1, 11, 12, 13, 2, 3 ]
